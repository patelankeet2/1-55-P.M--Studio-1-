CREATE TABLE STUDENT (
    S_ID INTEGER PRIMARY KEY AUTOINCREMENT,
    NAME TEXT NOT NULL,
    ADDRESS TEXT,
    DOB TEXT,
    CONTACT_INFO TEXT,
    ENROL_DATE TEXT,
    GRADE_LEVEL TEXT,
    ACAD_RECORDS TEXT,
    A_ID INTEGER,
    FOREIGN KEY (A_ID) REFERENCES ADMIN(A_ID)
);

CREATE TABLE TEACHER (
    T_ID INTEGER PRIMARY KEY AUTOINCREMENT,
    NAME TEXT NOT NULL,
    ADDRESS TEXT,
    SUB_EXPERT TEXT,
    CONTACT_INFO TEXT,
    PROF_DETAIL TEXT,
    A_ID INTEGER,
    FOREIGN KEY (A_ID) REFERENCES ADMIN(A_ID)
);
